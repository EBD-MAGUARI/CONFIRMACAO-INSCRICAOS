function compartilharWhatsApp() { // 1. Captura os dados diretamente dos elementos da tela que já foram preenchidos const nome = document.getElementById('d-nome').textContent; const telefone = document.getElementById('d-telefone').textContent; const congregacao = document.getElementById('d-congregacao').textContent; const grupo = document.getElementById('d-grupo').textContent; const area = document.getElementById('d-area').textContent; const pedido = document.getElementById('d-pedido').textContent; // 2. Usa a constante WHATSAPP_DESTINO que você já definiu no topo do script const urlZap = `https://api.whatsapp.com/send?phone=${WHATSAPP_DESTINO}&text=${encodeURIComponent(texto)}`; // 4.
